# ... so let's make The Ring ...

```javascript className=codehuge
const TheRing = (...args) => {
    const initial = proxify(args[0]);
    let current = initial;

    args.slice(1).forEach(arg => {
        const p = proxify(arg);
        Object.setPrototypeOf(current, p);
        current = p;
    }, current);

    Object.setPrototypeOf(current, initial);

    return initial;
};
```
